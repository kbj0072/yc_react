1-13. 콜백 함수

// 1. 콜백함수 (p. 51)
function main(value) {
  value(); // 콜백함수 실행
}

main(() => {
  console.log("i am sub");
}); // 콜백함수 전달

// 2. 콜백함수의 활용 (p. 54)
function repeat(count, callback) {
  for (let idx = 1; idx <= count; idx++) {
    callback(idx); // 콜백 함수에 1부터 count까지 파라미터로 넣어 반복 호출
  }
}

repeat(5, (idx) => { // 1부터 5까지 콜백함수(기존 값 출력) 반복 호출
  console.log(idx);
});

repeat(5, (idx) => { // 1부터 5까지 콜백함수(기존 값의 2배 출력) 반복 호출
  console.log(idx * 2);
});

repeat(5, (idx) => { // 1부터 5까지 콜백함수(기존 값의 3배 출력) 반복 호출
  console.log(idx * 3);
});
