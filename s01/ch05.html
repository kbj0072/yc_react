1-5. 자료형(타입)

// 1. 숫자형, Number Type (p. 19)
let num1 = 27;
let num2 = 1.5;
let num3 = -20;

let inf = Infinity;
let mInf = -Infinity;

let nan = NaN;

console.log(num1, num2, num3, inf, mInf, nan);

// 2. 문자형, String Type (p. 21)
let myName = "이정환";
let myLocation = "목동";
let introduce = myName + myLocation;
console.log(introduce);

let introduceText = `${myName}은 ${myLocation}에 거주합니다`;
console.log(introduceText);

// 3. Boolean Type (p. 22)
let isSwitchOn = true;
let isEmpty = false;
console.log(isSwitchOn, isEmpty);

// 4. Null Type (아무것도 없다)
let empty = null;
console.log(empty);

// 5. Undefined Type
let none;
console.log(none);
