1-11. 함수

// 함수 (p. 44)
let area1 = getArea(10, 20); // 내부에서 another 출력 후, 10 * 20 = 200 리턴
console.log(area1); // 200

let area2 = getArea(30, 20); // 내부에서 another 출력 후, 30 * 20 = 600 리턴
console.log(area2); // 600

getArea(120, 200); // 내부에서 another 출력 후, 120 * 200 = 24000 리턴(대입 없음)

// 호이스팅(Hoisting) - (p. 49)
// -> 끌어올리다 라는 뜻: 변수와 함수의 선언을 최상위로 끌어올리는 것
function getArea(width, height) {
  function another() {
    // 중첩 함수
    console.log("another");
  }

  another();
  let area = width * height;

  return area;
}
