1-12. 함수 표현식과 화살표 함수

// 1. 함수 표현식 (p. 50)
function funcA() {        // 함수 선언문, 호이스팅됨
  console.log("funcA");
}

let varA = funcA;
varA(); // 호출

let varB = function () {  // 함수 표현식, 익명함수를 저장
  console.log("funcB");
};

varB(); // 호출

// 2. 화살표 함수 (p. 55)
let varC = (value) => {
  console.log(value);
  return value + 1;
};

console.log(varC(10));
