2-7. 배열과 메서드 (1) - 요소 조작

// 6가지의 요소 조작 메서드 (p. 94)

// 1. push
// 배열의 맨 뒤에 새로운 요소를 추가하는 메서드
let arr1 = [1, 2, 3];
const newLength = arr1.push(4, 5, 6, 7);
console.log(arr1); // [1, 2, 3, 4, 5, 6, 7]

// 2. pop
// 배열의 맨 뒤에 있는 요소를 제거하고, 반환
let arr2 = [1, 2, 3];
const poppedItem = arr2.pop();
console.log(arr2); // [1, 2]

// 3. shift
// 배열의 맨 앞에 있는 요소를 제거, 반환
let arr3 = [1, 2, 3];
const shiftedItem = arr3.shift();
console.log(arr3); // [2, 3]

// 4. unshift
// 배열의 맨 앞에 새로운 요소를 추가하는 메서드
let arr4 = [1, 2, 3];
const newLength2 = arr4.unshift(0);
console.log(arr4); // [0, 1, 2, 3]

// shift와 unshift의 성능
// -> 배열의 모든 요소를 재배치하기 때문에 성능이 좋지 않음

// 5. slice
// 마치 가위처럼, 배열의 특정 범위를 잘라내서 새로운 배열로 반환
let arr5 = [1, 2, 3, 4, 5];
let sliced = arr5.slice(2, 5); // 2부터 5까지 잘라냄, 마지막 인덱스는 제외
let sliced2 = arr5.slice(2);  // 끝 인덱스를 생략하면 배열의 끝까지 잘라냄
let sliced3 = arr5.slice(-3); // 음수 인덱스를 사용하면 뒤에서부터 잘라냄
console.log(sliced);   // [3, 4, 5]
console.log(sliced2);  // [3, 4, 5]
console.log(sliced3);  // [3, 4, 5]

// 6. concat
// 두개의 서로 다른 배열을 이어 붙여서 새로운 배열을 반환
let arr6 = [1, 2];
let arr7 = [3, 4];

let concatedArr = arr6.concat(arr7);
console.log(concatedArr); // [1, 2, 3, 4]
