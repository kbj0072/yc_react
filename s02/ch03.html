2-3.구조 분해 할당

// 1. 배열의 구조 분해 할당 (p. 88)
let arr = [1, 2, 3];

let [one, two, three, four = 4] = arr;
console.log(one, two, three, four); // 1 2 3 4

// 2. 객체의 구조 분해 할당 (p. 89)
let person = {
  name: "이정환",
  age: 27,
  hobby: "테니스",
};

let {
  age: myAge, // age는 myAge로 이름 변경
  hobby,
  name,
  extra = "hello",
} = person; // age는 myAge로, hobby는 그대로, name은 그대로, extra는 "hello"로 기본값 설정
console.log(myAge, hobby, name, extra); // 27 테니스 이정환 hello

// 3. 함수의 매개변수가 객체일 때 구조 분해 할당하기 (p. 90)
const func = ({ name, age, hobby, extra }) => {
  console.log(name, age, hobby, extra);
};

func(person); // 이정환 27 테니스 undefined
