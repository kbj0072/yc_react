2-6.반복문 응용하기

// 1. 배열 순회 (p. 84)
let arr = [1, 2, 3, 4];

// 1.1 배열 인덱스
console.log("[인덱스를 이용한 순회]");
for (let i = 0; i < arr.length; i++) {
  console.log("\t", arr[i]);
}

// 1.2 for of 반복문 (p. 85)
// -> for of 문은 배열의 각 요소를 순회할 때 사용
console.log("[for...of 문을 이용한 순회]");
for (let item of arr) {
  console.log("\t", item);
}

// 2. 객체 순회 (p. 86)
let person = {
  name: "이정환",
  age: 27,
  hobby: "테니스",
};

// 2.1 Object.keys 사용
// -> 객체에서 key 값들만 뽑아서 새로운 배열로 반환
console.log("[Object.keys를 이용한 순회]");
let keys = Object.keys(person);

for (let key of keys) {
  const value = person[key];
  console.log("\t", key, "\t", value);
}

// 2.2 Object.values (p. 87)
// -> 객체에서 value 값들만 뽑아서 새로운 배열로 반환
console.log("[Object.values를 이용한 순회]");
let values = Object.values(person);

for (let value of values) {
  console.log("\t", value);
}

// 2.3 for in
// -> 객체의 모든 열거 가능한 속성(key)을 순회
console.log("[for...in 문을 이용한 순회]");
for (let key in person) {
  const value = person[key];
  console.log("\t", key, "\t", value);
}
